# Curl commands to test the Agentic AI and Sales AI components

# First, start the server (run this in a separate terminal):
# python main.py

# Assuming the server is running on http://127.0.0.1:8000

# 1. Test Sales Agent (default routing)
curl -X POST http://127.0.0.1:8000/sales \
  -H "Content-Type: application/json" \
  -d '{
    "message": "I am looking for a red dress",
    "user_id": 1,
    "session_id": "session123",
    "channel": "web",
    "history": []
